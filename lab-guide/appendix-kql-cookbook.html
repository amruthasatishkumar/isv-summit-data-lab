KQL Cookbook

Reusable Kusto patterns for the four streaming tables. Copy, paste, adapt. Every query here is read-only and uses table-scoped time windows for performance.

Universal Patterns

Most-recent reading per entity

KQL 
{TABLE}
| where timestamp > ago(15m)
| summarize arg_max(timestamp, *) by {ENTITY_KEY}

Replace {TABLE} with one of HospitalVitals, HospitalMovement, TrainTelemetry, FlightTelemetry, and {ENTITY_KEY} with patient_id, trainId, or hex.

Bin and chart over time

KQL 
{TABLE}
| where timestamp > ago(1h)
| summarize value = avg({METRIC}) by bin(timestamp, 1m)
| render timechart

Count rows per stream in last 5 minutes (health check)

KQL 
union withsource = table
    HospitalVitals, HospitalMovement, TrainTelemetry, FlightTelemetry
| where timestamp > ago(5m)
| summarize events = count() by table

🏥 HOSPITAL Patterns

Critical patients right now

KQL 
HospitalVitals
| where timestamp > ago(15m)
| summarize arg_max(timestamp, *) by patient_id
| where condition == "critical"
   or spo2 < 90
   or heart_rate > 120
   or heart_rate < 50
| project patient_id, condition, heart_rate, spo2, bp_systolic, bp_diastolic, timestamp

Patient location + latest vitals (join)

KQL 
let location =
    HospitalMovement
    | summarize arg_max(timestamp, *) by patient_id
    | where event_type != "Discharged"
    | project patient_id, current_location = to_location, last_event = event_type;
let vitals =
    HospitalVitals
    | summarize arg_max(timestamp, *) by patient_id;
location
| join kind = inner vitals on patient_id
| project patient_id, current_location, last_event, condition, heart_rate, spo2, timestamp

Room occupancy

KQL 
HospitalMovement
| summarize arg_max(timestamp, *) by patient_id
| where event_type != "Discharged"
| summarize patients = count() by to_location
| order by patients desc

SpO2 trend by patient (last 30 minutes)

KQL 
let critical_patients =
    HospitalVitals
    | where timestamp > ago(15m)
    | summarize arg_max(timestamp, *) by patient_id
    | where condition == "critical"
    | distinct patient_id;
HospitalVitals
| where timestamp > ago(30m)
| where patient_id in (critical_patients)
| summarize avg_spo2 = avg(spo2) by bin(timestamp, 1m), patient_id
| render timechart

Recent admissions

KQL 
HospitalMovement
| where timestamp > ago(1h)
| where event_type == "Admitted"
| project timestamp, patient_id, diagnosis_desc, to_location
| order by timestamp desc

🚆 TRANSIT Patterns

Current fleet status

KQL 
TrainTelemetry
| where timestamp > ago(10m)
| summarize arg_max(timestamp, *) by trainId
| project trainId, line, speed, status, lat, lon, timestamp

Average speed per line (last hour)

KQL 
TrainTelemetry
| where timestamp > ago(1h)
| summarize avg_speed = round(avg(speed), 1) by line
| order by avg_speed asc

Trains delayed in the last hour

KQL 
TrainTelemetry
| where timestamp > ago(1h)
| where status == "Delayed"
| summarize delayed_readings = count() by trainId, line
| order by delayed_readings desc

Speed timechart (per train)

KQL 
TrainTelemetry
| where timestamp > ago(1h)
| summarize avg_speed = avg(speed) by bin(timestamp, 1m), trainId
| render timechart

Map of current train positions

KQL 
TrainTelemetry
| where timestamp > ago(10m)
| summarize arg_max(timestamp, *) by trainId
| project trainId, lat, lon, line
| render scatterchart with (kind=map, ymin=lat, ymax=lat, xcolumn=lon)

✈️ AIRSPACE Patterns

Aircraft airborne right now

KQL 
FlightTelemetry
| where timestamp > ago(5m)
| summarize arg_max(timestamp, *) by hex
| where alt_baro > 0
| project flight, hex, lat, lon, alt_baro, gs, track, timestamp

Low-altitude (likely approach)

KQL 
FlightTelemetry
| where timestamp > ago(5m)
| summarize arg_max(timestamp, *) by hex
| where alt_baro between (0 .. 5000)
| project flight, alt_baro, gs, lat, lon
| order by alt_baro asc

Emergency squawks

KQL 
FlightTelemetry
| where timestamp > ago(15m)
| where squawk in ("7500", "7600", "7700")
| summarize arg_max(timestamp, *) by hex
| project flight, hex, squawk, alt_baro, lat, lon, timestamp

Aircraft within radius of a point

KQL 
let center_lat = 41.97;
let center_lon = -87.90;
let radius_km  = 50.0;
FlightTelemetry
| where timestamp > ago(5m)
| summarize arg_max(timestamp, *) by hex
| extend dist_km = geo_distance_2points(lon, lat, center_lon, center_lat) / 1000.0
| where dist_km < radius_km
| order by dist_km asc

Activity heatmap by altitude band

KQL 
FlightTelemetry
| where timestamp > ago(30m)
| summarize aircraft = dcount(hex) by bin(alt_baro, 5000)
| order by alt_baro asc
| render columnchart

Performance & Style Tips

  • Always lead with a where timestamp > ago(...) filter - Eventhouse is sharded by ingestion time.
  • Use arg_max(timestamp, *) by {key} instead of self-joins for "latest per entity".
  • Bin with bin(timestamp, 1m) for sub-hour trends; bin(timestamp, 1h) for daily trends.
  • render timechart needs a numeric value column + a time column. Categorical splits are auto-detected.
  • For RTI Dashboard tiles, parametrize the time range with _timeRange instead of hard-coding ago(15m).
  • Use summarize early. Filter early. Project late.
✓
The "right query" is usually the smallest one. Eventhouse rewards filtering and summarization, punishes returning million-row result sets. If a query takes > 2 seconds in this lab, it can probably be rewritten using arg_max or a tighter time window.
